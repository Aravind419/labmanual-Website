import 'package:flutter/material.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  // Root of the application üöÄ
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Best Widgets Demo',
      debugShowCheckedModeBanner: false, // Removes the debug label üö´üêû
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: BestWidgetsDemo(),
    );
  }
}

class BestWidgetsDemo extends StatefulWidget {
  @override
  _BestWidgetsDemoState createState() => _BestWidgetsDemoState();
}

class _BestWidgetsDemoState extends State {
  int _currentIndex = 0;
  double _sliderValue = 50.0;
  bool _toggleValue = false;

  // Three pages to showcase a variety of widgets
  late final List _pages;

  @override
  void initState() {
    super.initState();
    _pages = [
      _buildWidgetsDemoPage(),
      _buildListViewPage(),
      _buildGridViewPage(),
    ];
  }

  // Page 1: Various interactive widgets demo üé®
  Widget _buildWidgetsDemoPage() {
    return SingleChildScrollView(
      child: Padding(
        padding: const EdgeInsets.all(16.0), // Universal padding for comfort üìè
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.stretch,
          children: [
            Center(
              child: Text(
                "Welcome to the Best Widgets Demo",
                style: TextStyle(fontSize: 24, fontWeight: FontWeight.bold),
                textAlign: TextAlign.center,
              ),
            ),
            SizedBox(height: 20),
            Text(
              "This demo showcases interactive widgets like Cards, Sliders, Switches, Images, and more!",
              style: TextStyle(fontSize: 18),
              textAlign: TextAlign.center,
            ),
            SizedBox(height: 20),
            // A Card widget with ListTile and buttons
            Card(
              elevation: 4,
              child: Padding(
                padding: const EdgeInsets.all(16.0),
                child: Column(
                  children: [
                    ListTile(
                      leading: Icon(Icons.star, color: Colors.amber),
                      title: Text("Card Title"),
                      subtitle: Text("A subtitle that describes the content."),
                    ),
                    ButtonBar(
                      alignment: MainAxisAlignment.end,
                      children: [
                        TextButton(
                          onPressed: () {
                            ScaffoldMessenger.of(context).showSnackBar(
                              SnackBar(content: Text("Action 1 executed!")),
                            );
                          },
                          child: Text("ACTION 1"),
                        ),
                        TextButton(
                          onPressed: () {
                            ScaffoldMessenger.of(context).showSnackBar(
                              SnackBar(content: Text("Action 2 executed!")),
                            );
                          },
                          child: Text("ACTION 2"),
                        ),
                      ],
                    ),
                  ],
                ),
              ),
            ),
            SizedBox(height: 20),
            // Display an image using a network placeholder
            Center(
              child: Image.network(
                'https://via.placeholder.com/150',
                height: 150,
                width: 150,
              ),
            ),
            SizedBox(height: 20),
            // Slider widget to adjust a value
            Text(
              "Adjust Value: ${_sliderValue.toInt()}",
              style: TextStyle(fontSize: 18),
              textAlign: TextAlign.center,
            ),
            Slider(
              value: _sliderValue,
              min: 0,
              max: 100,
              onChanged: (value) {
                setState(() {
                  _sliderValue = value;
                });
              },
            ),
            SizedBox(height: 20),
            // Toggle switch widget for a boolean value
            Row(
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                Text("Toggle Option", style: TextStyle(fontSize: 18)),
                Switch(
                  value: _toggleValue,
                  onChanged: (value) {
                    setState(() {
                      _toggleValue = value;
                    });
                  },
                ),
              ],
            ),
          ],
        ),
      ),
    );
  }

  // Page 2: ListView with Cards and ListTiles üìú
  Widget _buildListViewPage() {
    return ListView.builder(
      itemCount: 10,
      itemBuilder: (context, index) {
        return Card(
          elevation: 2,
          margin: EdgeInsets.symmetric(horizontal: 16, vertical: 8),
          child: ListTile(
            leading: Icon(Icons.label),
            title: Text("List Item ${index + 1}"),
            subtitle: Text("Subtitle for item ${index + 1}"),
            trailing: Icon(Icons.arrow_forward),
            onTap: () {
              ScaffoldMessenger.of(context).showSnackBar(
                SnackBar(content: Text("Tapped on item ${index + 1}")),
              );
            },
          ),
        );
      },
    );
  }

  // Page 3: GridView demonstrating a grid of Cards üî≤
  Widget _buildGridViewPage() {
    return Padding(
      padding: const EdgeInsets.all(16.0),
      child: GridView.builder(
        itemCount: 8,
        gridDelegate: SliverGridDelegateWithFixedCrossAxisCount(
          crossAxisCount: 2, // Two columns for grid layout
          crossAxisSpacing: 10,
          mainAxisSpacing: 10,
        ),
        itemBuilder: (context, index) {
          return Card(
            elevation: 4,
            child: Center(
              child: Text(
                "Grid Item ${index + 1}",
                style: TextStyle(fontSize: 18),
              ),
            ),
          );
        },
      ),
    );
  }

  // Main Scaffold with Drawer, BottomNavigationBar, FloatingActionButton, and BottomSheet
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("Best Widgets Demo"),
        centerTitle: true,
      ),
      drawer: Drawer(
        child: ListView(
          children: [
            DrawerHeader(
              decoration: BoxDecoration(color: Colors.blue),
              child: Center(
                child: Text(
                  "Menu",
                  style: TextStyle(color: Colors.white, fontSize: 24),
                ),
              ),
            ),
            ListTile(
              leading: Icon(Icons.widgets),
              title: Text("Widgets Demo"),
              onTap: () {
                Navigator.pop(context);
                setState(() {
                  _currentIndex = 0;
                });
              },
            ),
            ListTile(
              leading: Icon(Icons.list),
              title: Text("List View"),
              onTap: () {
                Navigator.pop(context);
                setState(() {
                  _currentIndex = 1;
                });
              },
            ),
            ListTile(
              leading: Icon(Icons.grid_on),
              title: Text("Grid View"),
              onTap: () {
                Navigator.pop(context);
                setState(() {
                  _currentIndex = 2;
                });
              },
            ),
          ],
        ),
      ),
      body: _pages[_currentIndex],
      bottomNavigationBar: BottomNavigationBar(
        currentIndex: _currentIndex,
        onTap: (int index) {
          setState(() {
            _currentIndex = index;
          });
        },
        items: [
          BottomNavigationBarItem(icon: Icon(Icons.widgets), label: "Widgets"),
          BottomNavigationBarItem(icon: Icon(Icons.list), label: "List"),
          BottomNavigationBarItem(icon: Icon(Icons.grid_on), label: "Grid"),
        ],
      ),
      floatingActionButton: FloatingActionButton(
        onPressed: () {
          ScaffoldMessenger.of(context).showSnackBar(
            SnackBar(content: Text("Floating Action Button Pressed!")),
          );
        },
        child: Icon(Icons.add),
      ),
      bottomSheet: Container(
        width: double.infinity,
        color: Colors.grey[200],
        padding: EdgeInsets.all(8),
        child: Center(
          child: Text(
            "¬© Aravind reserved a like",
            style: TextStyle(fontSize: 14),
          ),
        ),
      ),
    );
  }
}
